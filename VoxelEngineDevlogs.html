Voxel Engine

A Voxel Renderer made in C/C++ with OpenGL

- 12/21/2023
- Hello World!

I will be denoting my C++ code in purple, my C code in blue, and my miscelenous (shaders, CMake, etc.) in orange. The first thing I had to do was setup my my CMake files and a basic hello world file to prove everything was included and compiling correctly. 

Top-Level CMakeLists.txt

cmake_minimum_required (VERSION 3.8)

# Enable Hot Reload for MSVC compilers if supported.
if (POLICY CMP0141)
    cmake_policy(SET CMP0141 NEW)
    set(CMAKE_MSVC_DEBUG_INFORMATION_FORMAT "$,$>,$<$:EditAndContinue>,$<$:ProgramDatabase>>")
endif()


include_directories(source_files PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/dependencies/include)
link_directories(source_files PRIVATE ${CMAKE_CURRENT_SOURCE_DIR}/dependencies/lib-mingw-w64)

project("VoxelEngine")

add_subdirectory("src")
